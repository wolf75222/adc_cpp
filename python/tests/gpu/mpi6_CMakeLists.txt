cmake_minimum_required(VERSION 3.16)
project(mpi6 CXX)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(Kokkos REQUIRED)
find_package(MPI REQUIRED)
add_executable(mpi6_fillboundary mpi6_fillboundary.cpp)
target_include_directories(mpi6_fillboundary PRIVATE ${POPS_INCLUDE})
target_compile_definitions(mpi6_fillboundary PRIVATE POPS_HAS_KOKKOS POPS_HAS_MPI)
target_compile_features(mpi6_fillboundary PRIVATE cxx_std_20)
target_link_libraries(mpi6_fillboundary PRIVATE Kokkos::kokkos MPI::MPI_CXX)
