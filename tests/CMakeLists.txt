# Tests sans dependance externe pour ce premier socle (executable + CTest).
# Catch2 sera cable quand la surface de test le justifiera.

function(adc_add_test name)
  add_executable(${name} ${name}.cpp)
  target_link_libraries(${name} PRIVATE adc::adc)
  add_test(NAME ${name} COMMAND ${name})
endfunction()

# Test MPI : seulement si le backend MPI est active. Lance via mpirun -np 4 ;
# le seam (rang/size, all-reduce) doit donner le meme resultat quel que soit N.
if(ADC_HAS_MPI)
  function(adc_add_mpi_test name nproc)
    add_executable(${name} ${name}.cpp)
    target_link_libraries(${name} PRIVATE adc::adc)
    add_test(NAME ${name}
             COMMAND ${MPIEXEC_EXECUTABLE} ${MPIEXEC_NUMPROC_FLAG} ${nproc}
                     ${MPIEXEC_PREFLAGS} $<TARGET_FILE:${name}>
                     ${MPIEXEC_POSTFLAGS})
  endfunction()
  adc_add_mpi_test(test_mpi_smoke 4)
  adc_add_mpi_test(test_mpi_fillboundary 4)
  adc_add_mpi_test(test_mpi_advect 4)
  adc_add_mpi_test(test_mpi_poisson 4)
  adc_add_mpi_test(test_mpi_diocotron 4)
  adc_add_mpi_test(test_mpi_redistribute 4)
  adc_add_mpi_test(test_mpi_overlap 4)
endif()
